216 /* Debug output macros. */
217 #define DEBUG_FMT(fmt, ...) do { \
218  if (sp_debug_handler) \
219  sp_debug_handler(fmt ".\n", __VA_ARGS__); \
220 } while (0)
221 #define DEBUG(msg) DEBUG_FMT(msg, NULL)
222 #define DEBUG_ERROR(err, msg) DEBUG_FMT("%s returning " #err ": " msg, __func__)
223 #define DEBUG_FAIL(msg) do { \
224  char *errmsg = sp_last_error_message(); \
225  DEBUG_FMT("%s returning SP_ERR_FAIL: " msg ": %s", __func__, errmsg); \
226  sp_free_error_message(errmsg); \
227 } while (0);
228 #define RETURN() do { \
229  DEBUG_FMT("%s returning", __func__); \
230  return; \
231 } while (0)
232 #define RETURN_CODE(x) do { \
233  DEBUG_FMT("%s returning " #x, __func__); \
234  return x; \
235 } while (0)
236 #define RETURN_CODEVAL(x) do { \
237  switch (x) { \
238  case SP_OK: RETURN_CODE(SP_OK); \
239  case SP_ERR_ARG: RETURN_CODE(SP_ERR_ARG); \
240  case SP_ERR_FAIL: RETURN_CODE(SP_ERR_FAIL); \
241  case SP_ERR_MEM: RETURN_CODE(SP_ERR_MEM); \
242  case SP_ERR_SUPP: RETURN_CODE(SP_ERR_SUPP); \
243  default: RETURN_CODE(SP_ERR_FAIL); \
244  } \
245 } while (0)
246 #define RETURN_OK() RETURN_CODE(SP_OK);
247 #define RETURN_ERROR(err, msg) do { \
248  DEBUG_ERROR(err, msg); \
249  return err; \
250 } while (0)
251 #define RETURN_FAIL(msg) do { \
252  DEBUG_FAIL(msg); \
253  return SP_ERR_FAIL; \
254 } while (0)
255 #define RETURN_INT(x) do { \
256  int _x = x; \
257  DEBUG_FMT("%s returning %d", __func__, _x); \
258  return _x; \
259 } while (0)
260 #define RETURN_STRING(x) do { \
261  char *_x = x; \
262  DEBUG_FMT("%s returning %s", __func__, _x); \
263  return _x; \
264 } while (0)
265 #define RETURN_POINTER(x) do { \
266  void *_x = x; \
267  DEBUG_FMT("%s returning %p", __func__, _x); \
268  return _x; \
269 } while (0)
270 #define SET_ERROR(val, err, msg) do { DEBUG_ERROR(err, msg); val = err; } while (0)
271 #define SET_FAIL(val, msg) do { DEBUG_FAIL(msg); val = SP_ERR_FAIL; } while (0)
272 #define TRACE(fmt, ...) DEBUG_FMT("%s(" fmt ") called", __func__, __VA_ARGS__)
273 #define TRACE_VOID() DEBUG_FMT("%s() called", __func__)
